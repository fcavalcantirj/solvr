    **backend/internal/api/middleware/bearer_guard.go:**

    Create middleware that extracts a room bearer token from the Authorization header, hashes it with `token.HashToken()`, looks up the room via `RoomRepository.GetByTokenHash()`, and injects the resolved `*models.Room` into the request context.

    ```go
    package middleware

    import (
        "context"
        "net/http"
        "strings"

        "github.com/fcavalcantirj/solvr/internal/db"
        "github.com/fcavalcantirj/solvr/internal/token"
    )

    type contextKey string
    const RoomContextKey contextKey = "room"

    func RoomFromContext(ctx context.Context) *models.Room {
        room, _ := ctx.Value(RoomContextKey).(*models.Room)
        return room
    }

    func BearerGuard(roomRepo *db.RoomRepository) func(http.Handler) http.Handler {
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                authHeader := r.Header.Get("Authorization")
                if !strings.HasPrefix(authHeader, "Bearer ") {
                    // Also check query param ?token= for SSE connections (browsers can't set headers on EventSource)
                    tokenParam := r.URL.Query().Get("token")
                    if tokenParam == "" {
                        http.Error(w, `{"error":{"code":"UNAUTHORIZED","message":"missing bearer token"}}`, http.StatusUnauthorized)
                        return
                    }
                    authHeader = "Bearer " + tokenParam
                }
                plaintext := strings.TrimPrefix(authHeader, "Bearer ")
                hash := token.HashToken(plaintext)
                room, err := roomRepo.GetByTokenHash(r.Context(), hash)
                if err != nil {
                    http.Error(w, `{"error":{"code":"UNAUTHORIZED","message":"invalid room token"}}`, http.StatusUnauthorized)
                    return
                }
                ctx := context.WithValue(r.Context(), RoomContextKey, room)
                next.ServeHTTP(w, r.WithContext(ctx))
            })
        }
    }
    ```
